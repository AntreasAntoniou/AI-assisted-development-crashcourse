import pathlib
import fire
from rich.progress import track
from rich.console import Console

console = Console()

def ingest_content(
    root_dir: str,
    output_file: str,
    extensions: list[str] = ['.py', '.md', '.html', '.css', '.js'],
):
    """
    Ingests content from files with specified extensions from a root directory
    into a single text file.

    Args:
        root_dir (str): The root directory to search for files.
        output_file (str): The path to the output text file.
        extensions (list[str]): A list of file extensions to include.
    """
    root_path = pathlib.Path(root_dir)
    output_path = pathlib.Path(output_file)
    
    console.print(f"üîç Starting ingestion from [bold cyan]{root_path}[/bold cyan]...")
    
    # First, find all relevant files
    files_to_ingest = []
    for ext in extensions:
        files_to_ingest.extend(list(root_path.rglob(f"*{ext}")))
        
    if not files_to_ingest:
        console.print("[bold red]No files found with the specified extensions.[/bold red]")
        return

    console.print(f"Found [bold green]{len(files_to_ingest)}[/bold green] files to ingest.")

    with open(output_path, 'w', encoding='utf-8') as outfile:
        for file_path in track(files_to_ingest, description="Ingesting files..."):
            try:
                with open(file_path, 'r', encoding='utf-8', errors='ignore') as infile:
                    content = infile.read()
                    outfile.write(f"--- Start of {file_path.relative_to(root_path)} ---\n\n")
                    outfile.write(content)
                    outfile.write(f"\n\n--- End of {file_path.relative_to(root_path)} ---\n\n")
            except Exception as e:
                console.print(f"[bold red]Error reading {file_path}: {e}[/bold red]")
    
    console.print(f"‚úÖ Successfully ingested content into [bold cyan]{output_path}[/bold cyan].")

if __name__ == '__main__':
    fire.Fire(ingest_content)

import pathlib
import fire
import ast
from rich.console import Console

console = Console()

def _get_signature(node):
    """Extracts a simplified signature from a function or class AST node."""
    if isinstance(node, ast.FunctionDef) or isinstance(node, ast.AsyncFunctionDef):
        args = [arg.arg for arg in node.args.args]
        return f"{node.name}({', '.join(args)})"
    elif isinstance(node, ast.ClassDef):
        return f"class {node.name}:"
    return ""

def map_directory_structure(
    root_dir: str,
    output_file: str,
    ignore_dirs: list[str] = None,
):
    """
    Generates a markdown file representing the directory structure,
    including classes and functions from Python files.

    Args:
        root_dir (str): The root directory to map.
        output_file (str): The path to the output markdown file.
        ignore_dirs (list[str]): A list of directory names to ignore.
    """
    root_path = pathlib.Path(root_dir)
    output_path = pathlib.Path(output_file)
    if ignore_dirs is None:
        ignore_dirs = ['__pycache__', '.git', '.vscode', 'venv', '.idea']

    console.print(f"üó∫Ô∏è  Mapping directory structure of [bold cyan]{root_path}[/bold cyan]...")

    with open(output_path, 'w', encoding='utf-8') as outfile:
        outfile.write(f"# Directory Map of {root_path.name}\n\n")

        def recurse_path(path, prefix=""):
            # Sort items, directories first
            items = sorted(list(path.iterdir()), key=lambda p: not p.is_dir())
            
            for i, item in enumerate(items):
                is_last = (i == len(items) - 1)
                connector = "‚îî‚îÄ‚îÄ " if is_last else "‚îú‚îÄ‚îÄ "

                if item.is_dir():
                    if item.name in ignore_dirs:
                        continue
                    outfile.write(f"{prefix}{connector}{item.name}/\n")
                    new_prefix = prefix + ("    " if is_last else "‚îÇ   ")
                    recurse_path(item, new_prefix)
                else:
                    outfile.write(f"{prefix}{connector}{item.name}\n")
                    if item.suffix == '.py':
                        try:
                            with open(item, 'r', encoding='utf-8') as py_file:
                                content = py_file.read()
                                tree = ast.parse(content)
                                new_prefix = prefix + ("    " if is_last else "‚îÇ   ")
                                for node in ast.walk(tree):
                                    if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef, ast.ClassDef)):
                                        # We only want top-level functions/classes
                                        if isinstance(node, ast.ClassDef) and node.col_offset == 0:
                                            outfile.write(f"{new_prefix}  - {_get_signature(node)}\n")
                                        elif isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)) and node.col_offset == 0:
                                             outfile.write(f"{new_prefix}  - {_get_signature(node)}\n")

                        except Exception as e:
                            outfile.write(f"{new_prefix}  - (Could not parse: {e})\n")

        recurse_path(root_path)

    console.print(f"‚úÖ Successfully created directory map at [bold cyan]{output_path}[/bold cyan].")

if __name__ == '__main__':
    fire.Fire(map_directory_structure)

    -   **Simulated Understanding:** The "understanding" you perceive is an emergent property of these patterns. This is why psychological cues work. By providing context, you guide the model toward a better statistical neighborhood.
    -   **Finding the Vibe (The Art of Context):** Think of it like tuning a radio or finding a dance partner. Your goal is to provide the perfect context to lead the AI to the right answer. This includes:

### **Slide 12: Why AI Chats "Age"**
-   **Title:** Why AI Chats "Age"
-   **Content:**
    -   **The Golden Age:** At the start of a chat, the context is small and clean. The AI is highly focused and effective.

-   **Content:**

        4.  You say "fix the new failing tests." The AI, now believing the feature is wrong, continues to "fix" or delete other features.

### 1. The First-Principles Thinker

*   **Philosophy:** Deconstructs problems to their most fundamental truths, questioning every assumption to enable radical innovation.

            <div class="content-box text-2xl space-y-4 text-left">
                <p contenteditable="true"><strong>The Golden Age:</strong> At the start of a chat, the context is small and clean. The AI is highly focused and effective.</p>
                <p contenteditable="true"><strong>The "Lost in the Middle" Problem:</strong> As the chat history grows, models tend to pay more attention to the very beginning and the very end of the conversation. This is a known architectural limitation, and critical details mentioned in the middle can get "lost" or ignored.</p>
                <p contenteditable="true"><strong>Catastrophic Forgetting:</strong> Eventually, the context becomes so large and noisy that performance degrades significantly. The AI forgets key constraints or reverts to generic behavior. This is your cue to start a new session.</p>

                <p contenteditable="true"><strong>The "Context Carry-Over" Technique:</strong></p>
                <ol class="list-decimal list-inside text-xl space-y-2 mt-2">
                    <li contenteditable="true">Use the <code>Export Chat</code> feature to get a full transcript of your conversation.</li>
                    <li contenteditable="true">Open the transcript and manually copy the most critical pieces of context: key decisions, final code snippets, and important constraints.</li>
                    <li contenteditable="true"><strong>This curation is the most important step.</strong> A small, highly-relevant context is far more powerful than a large, noisy one.</li>
                    <li contenteditable="true">Paste this curated context as the very first prompt in your new chat session to get the AI up to speed instantly.</li>

                <p contenteditable="true">The entire codebase might be too large to fit in the context window.</p>

        <!-- Slide 16: Scoping: Large Projects -->
        <div class="slide">
            <h2 class="text-5xl font-bold gradient-text mb-8" contenteditable="true">Large-Scale & Legacy Projects</h2>
            <div class="content-box text-left text-xl space-y-3">
                <p class="text-2xl text-center" contenteditable="true"><strong>Context Strategy:</strong> The "Directory Map" Approach.</p>
                <p class="text-center">Full ingestion is impossible and undesirable.</p>
                <p class="mt-4"><strong>The Strategy:</strong> Use a script to map the entire directory structure, including function and class signatures from key files, into a single markdown file (<code>directory_map.md</code>).</p>
                <p class="mt-2"><strong>Workflow:</strong></p>
                <ol class="list-decimal list-inside mt-2 space-y-2">
                    <li contenteditable="true">Start your session by giving the AI the <code>directory_map.md</code>. It now has a high-level "map" of the entire codebase.</li>
                    <li contenteditable="true">When you need to work on a specific feature, ask the AI: "Based on the map, which files are most relevant for implementing X?"</li>
                    <li contenteditable="true">Add only those specific files to the context. This allows you to navigate massive codebases effectively.</li>
                </ol>
            </div>

            <div class="content-box text-2xl space-y-4 text-left">
                <ol class="list-decimal list-inside space-y-3">
                    <li contenteditable="true"><strong>Discuss the Feature:</strong> Massage the ideas back and forth. Crucially, tell the AI: <strong>"Do not act until I approve the plan."</strong></li>
                    <li contenteditable="true"><strong>Demand a Plan:</strong> Ask for a plan, architecture, and stack if it's complex.</li>
                    <li contenteditable="true"><strong>Create Milestones:</strong> Have the AI create a <code>features_built/feature_name/milestones.md</code> file to track its own progress.</li>
                </ol>
            </div>

        <!-- Slide 22: Phase 2: Test-Driven Development -->
        <div class="slide">
            <h2 class="text-5xl font-bold gradient-text mb-8" contenteditable="true">Phase 2: Test-Driven Development</h2>
            <div class="content-box text-2xl space-y-4 text-left">
                <ol class="list-decimal list-inside space-y-3">
                    <li contenteditable="true"><strong>Define Interfaces & Write Tests First:</strong> Decide on the function/class interfaces <em>before</em> implementation. Then, ask the AI to write tests.</li>
                    <li contenteditable="true"><strong>Skim-Review the Tests:</strong> Ensure the tests cover the main use cases and edge cases. This is your safety net.</li>
                </ol>
            </div>
        </div>

        <!-- Slide 23: Phase 3: Implementation & Verification -->
        <div class="slide">
            <h2 class="text-5xl font-bold gradient-text mb-8" contenteditable="true">Phase 3: Implementation & Verification</h2>
            <div class="content-box text-2xl space-y-4 text-left">
                <ol class="list-decimal list-inside space-y-3">
                    <li contenteditable="true"><strong>Step-by-Step Implementation:</strong> Ask the AI to develop code one step at a time. <strong>DO NOT SKIP THE REVIEW.</strong></li>
                    <li class="p-4 bg-red-900/50 border-l-4 border-red-500 rounded" contenteditable="true"><strong>Warning:</strong> Blindly accepting code is how you lose work. The AI may "fix" a trivial issue by deleting a complex module if left unsupervised.</li>
                    <li contenteditable="true"><strong>Run ALL Tests:</strong> After every significant change, run the full test suite. Repeat until all tests pass.</li>
                </ol>
            </div>
        </div>

        <!-- Slide 24: Warning: The "Ground Truth" Spiral -->
        <div class="slide">
            <h2 class="text-5xl font-bold gradient-text mb-8" contenteditable="true">Warning: The "Ground Truth" Spiral</h2>
            <div class="content-box text-2xl space-y-4 text-left">
                <p>This is a catastrophic failure mode. It begins when a test fails and the AI is unsure what is correct: the test or the feature.</p>
                <p class="mt-4"><strong>The Spiral:</strong></p>
                <ol class="text-xl space-y-2 mt-2 list-decimal list-inside">
                    <li contenteditable="true">You tell the AI "fix the tests."</li>
                    <li contenteditable="true">The AI sees a flawed test and <strong>changes your working feature code to be incorrect</strong> to satisfy the bad test.</li>
                    <li contenteditable="true">This change causes other, correct tests to fail.</li>
                    <li contenteditable="true">You say "fix the new failing tests." The AI, now believing the feature is wrong, continues to "fix" or delete other features.</li>
                </ol>
                <p class="mt-4 p-4 bg-red-900/50 border-l-4 border-red-500 rounded"><strong>You are the Arbiter of Truth.</strong> You must explicitly tell the AI which part is correct.</p>
            </div>
        </div>

        <!-- Slide 25: Phase 4: Finalization -->
        <div class="slide">
            <h2 class="text-5xl font-bold gradient-text mb-8" contenteditable="true">Phase 4: Finalization</h2>
            <div class="content-box text-2xl space-y-4 text-left">
                <ol class="list-decimal list-inside space-y-3">
                    <li contenteditable="true"><strong>Verify with Examples:</strong> Have the AI write simple execution examples (e.g., in <code>if __name__ == "__main__":</code>). Run them and inspect the output together.</li>
                    <li contenteditable="true"><strong>Document and Comment:</strong> Once everything is working, ask the AI to thoroughly document the code.</li>
                </ol>
            </div>
